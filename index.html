// 게임 오버 오버레이 제거 const existingOverlays = gameArea.querySelectorAll('.game-over-animation'); existingOverlays.forEach(overlay => overlay.remove()); // 게임 요소 초기화 const items = document.querySelectorAll('.falling-item'); items.forEach(item => item.remove()); // 버튼 상태 초기화 document.getElementById('appleCashOutBtn').classList.add('hidden'); document.getElementById('applePlayAgainBtn').classList.add('hidden'); // 게임 준비 오버레이 표시 document.getElementById('gameOverlay').classList.remove('hidden'); // 바구니 드래그 기능 설정 setupBasketDrag(); } // 바구니 드래그 이벤트 설정 function setupBasketDrag() { // 기존 이벤트 제거 basketElement.removeEventListener('mousedown', startDragging); document.removeEventListener('mousemove', dragBasket); document.removeEventListener('mouseup', stopDragging); basketElement.removeEventListener('touchstart', startDragging); document.removeEventListener('touchmove', dragBasket); document.removeEventListener('touchend', stopDragging); // 새 이벤트 등록 basketElement.addEventListener('mousedown', startDragging); document.addEventListener('mousemove', dragBasket); document.addEventListener('mouseup', stopDragging); basketElement.addEventListener('touchstart', startDragging); document.addEventListener('touchmove', dragBasket); document.addEventListener('touchend', stopDragging); } // 드래그 시작 function startDragging(e) { if (!appleGameActive) return; isDragging = true; basketElement.style.cursor = 'grabbing'; if (e.type === 'mousedown') { startX = e.clientX; lastX = startX; } else if (e.type === 'touchstart') { startX = e.touches[0].clientX; lastX = startX; // 터치 이벤트에서는 화면 스크롤 방지 e.preventDefault(); } } // 드래그 중 function dragBasket(e) { if (!isDragging || !appleGameActive) return; let currentX; if (e.type === 'mousemove') { currentX = e.clientX; } else if (e.type === 'touchmove') { currentX = e.touches[0].clientX; // 터치 이벤트에서는 화면 스크롤 방지 e.preventDefault(); } // 이동 거리 계산 const deltaX = currentX - lastX; lastX = currentX; // 게임 영역의 너비 계산 const gameAreaRect = gameArea.getBoundingClientRect(); const gameAreaWidth = gameAreaRect.width; // 이동 거리를 퍼센트로 변환 const movePercent = (deltaX / gameAreaWidth) * 100; // 바구니 위치 업데이트 basketPosition += movePercent; // 바구니가 화면 밖으로 나가지 않도록 제한 basketPosition = Math.max(5, Math.min(90, basketPosition)); // 바구니 위치 적용 basketElement.style.left = `${basketPosition}%`; } // 드래그 종료 function stopDragging() { isDragging = false; basketElement.style.cursor = 'grab'; } document.getElementById('startGameBtn').addEventListener('click', function() { // 게임 준비 오버레이 숨기기 document.getElementById('gameOverlay').classList.add('hidden'); // 게임 활성화 appleGameActive = true; // 배경 음악 재생 (사용자 상호작용으로 재생) try { if (!isMuted) { bgMusic.volume = 0.5; let playPromise = bgMusic.play(); // 브라우저 호환성을 위한 처리 if (playPromise !== undefined) { playPromise.catch(error => { console.log("음악 재생 실패:", error); // 실패해도 게임은 계속 진행 }); } } } catch (e) { console.log("오디오 재생 오류:", e); } // 타이머 시작 startGameTimer(); // 아이템 생성 시작 gameInterval = setInterval(generateFallingItem, 600); // 0.6초마다 아이템 생성 // 충돌 감지 업데이트 시작 collisionInterval = setInterval(checkCollisions, 20); }); function startGameTimer() { const timerInterval = setInterval(() => { gameTimeLeft--; document.getElementById('gameTimer').textContent = `${gameTimeLeft}초`; // 시간이 다 되면 게임 종료 if (gameTimeLeft <= 0) { clearInterval(timerInterval); clearInterval(gameInterval); clearInterval(collisionInterval); // 배경음악 정지 if (bgMusic) bgMusic.pause(); // 성공 효과음 (내장 효과음 사용) playSuccessSound(); endGame(); } }, 1000); } function generateFallingItem() { if (!appleGameActive) return; // 한번에 2~3개의 아이템 생성 (더 많이) const itemCount = Math.floor(Math.random() * 2) + 2; for (let i = 0; i < itemCount; i++) { const item = document.createElement('div'); // 벌레 비율 증가: 30% 사과, 70% 벌레 const isApple = Math.random() < 0.3; item.textContent = isApple ? '🍎' : '🐛'; item.className = 'falling-item text-2xl'; item.dataset.type = isApple ? 'apple' : 'bug'; // 랜덤 위치 (좌우) const left = Math.floor(Math.random() * 85) + 5; // 5% ~ 90% item.style.left = `${left}%`; // 랜덤 속도 (아래로 떨어지는 시간) const fallSpeed = Math.random() * 1 + 1.2; // 1.2초 ~ 2.2초 (약간 더 빠르게) item.style.animationDuration = `${fallSpeed}s`; gameArea.appendChild(item); // 바닥에 닿으면 제거 item.addEventListener('animationend', () => { if (!item.dataset.collected) { // 아직 수집되지 않은 경우에만 item.remove(); } }); } } // 충돌 감지 함수 function checkCollisions() { if (!appleGameActive) return; const items = document.querySelectorAll('.falling-item'); const gameAreaRect = gameArea.getBoundingClientRect(); // 바구니 위치 계산 (바구니 상단 부분에서 감지) const basketLeft = basketPosition - basketWidth/2; const basketRight = basketPosition + basketWidth/2; const basketTop = 85; // 바구니 상단 위치 (%) items.forEach(item => { // 이미 처리된 아이템인지 확인 if (item.dataset.collected === "true") return; // 아이템의 위치 계산 const itemRect = item.getBoundingClientRect(); const itemLeft = parseFloat(item.style.left); const itemTop = (itemRect.top - gameAreaRect.top) / gameAreaRect.height * 100; // 바구니와 아이템 충돌 감지 (단순 위치 기반) if (itemTop >= basketTop && itemLeft >= basketLeft && itemLeft <= basketRight) { // 처리된 것으로 표시 item.dataset.collected = "true"; // 즉시 아이템 애니메이션 중지 (떨어지는 것을 멈추고 사라지게) item.style.animationPlayState = 'paused'; if (item.dataset.type === 'apple') { // 사과 획득 appleScore++; document.getElementById('appleScore').textContent = appleScore; // 사과 획득 효과음 재생 (내장 효과음 사용) playAppleSound(); // 점수 증가 애니메이션 const scoreElement = document.getElementById('appleScore'); scoreElement.classList.add('score-pop'); setTimeout(() => { scoreElement.classList.remove('score-pop'); }, 500); // 사과 사라지는 효과 (즉시 더 투명하게 시작) item.style.opacity = "0"; item.style.transform = "scale(0.5)"; setTimeout(() => { item.remove(); }, 100); // 더 빠르게 제거 } else { // 벌레 먹음 - 게임 즉시 종료! // 배경음악 정지 if (bgMusic) bgMusic.pause(); // 게임 오버 효과음 재생 (내장 효과음 사용) playGameOverSound(); // 벌레 사라지는 효과 (즉시 더 투명하게 시작) item.style.opacity = "0"; item.style.transform = "scale(0.5)"; // 바구니 흔들림 애니메이션 basketElement.classList.add('shake'); // 게임 오버 효과 showGameOverEffect(); // 약간의 지연 후 게임 종료 (효과를 볼 수 있도록) setTimeout(() => { endGame(true); // true = 벌레에 의한 게임 오버 }, 500); } } }); } // 게임 오버 효과 function showGameOverEffect() { // 게임 활성화 상태 비활성화 appleGameActive = false; // 게임 영역에 붉은 오버레이 추가 const gameOverlay = document.createElement('div'); gameOverlay.className = 'absolute inset-0 bg-red-500 bg-opacity-50 flex items-center justify-center z-10 game-over-animation'; // 게임 오버 메시지 const messageDiv = document.createElement('div'); messageDiv.className = 'bg-white p-3 rounded-lg text-center'; messageDiv.innerHTML = '
게임 오버!
벌레를 먹었습니다!
'; gameOverlay.appendChild(messageDiv); gameArea.appendChild(gameOverlay); } function endGame(isGameOver = false) { appleGameActive = false; // 인터벌 정리 clearInterval(gameInterval); clearInterval(collisionInterval); // 배경 음악 정지 bgMusic.pause(); if (!isGameOver) { // 정상 종료 (시간 초과) // 골드 계산 (사과 1개당 3골드) appleGold = appleScore * 3; // 상금 한도 설정 appleGold = Math.min(appleGold, 50); // UI 업데이트 document.getElementById('appleGold').textContent = appleGold; document.getElementById('appleGameResult').classList.remove('hidden'); document.getElementById('finalAppleGold').textContent = appleGold; // 골드 증가 const gold = document.getElementById('gold'); gold.textContent = parseInt(gold.textContent) + appleGold; // 알림 표시 showToast(`${appleGold} 골드가 지급되었습니다!`, 'success'); } else { // 게임 오버 (벌레 먹음) // 결과 메시지 표시 document.getElementById('appleGameResult').classList.remove('hidden'); document.getElementById('appleGameResult').querySelector('div').innerHTML = `
게임 오버! 벌레를 먹었습니다.
획득한 사과: ${appleScore}개
`; } // 다시하기 버튼 표시 document.getElementById('applePlayAgainBtn').classList.remove('hidden'); } // 사과 게임 뒤로가기 버튼 document.getElementById('appleBackBtn').addEventListener('click', () => { // 게임 중이라면 종료 if (appleGameActive) { clearInterval(gameInterval); clearInterval(collisionInterval); // 배경 음악 정지 bgMusic.pause(); } switchTab('homeTab'); }); // 사과 게임 다시하기 버튼 document.getElementById('applePlayAgainBtn').addEventListener('click', () => { // 광고 시청 후 다시하기 watchAd(() => { initAppleGame(); }); }); // 상점 아이템 구매 기능 const shopItems = document.querySelectorAll('.shop-item'); shopItems.forEach(item => { item.addEventListener('click', function() { const price = parseInt(this.getAttribute('data-price')); const statType = this.getAttribute('data-stat'); const statValue = parseInt(this.getAttribute('data-value')); // 현재 골드 확인 const gold = document.getElementById('gold'); const currentGold = parseInt(gold.textContent);
